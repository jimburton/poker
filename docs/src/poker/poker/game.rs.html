24/// Enum for representing the stage of a round.
25#[derive(Debug, PartialEq, Clone, Copy, Serialize, Deserialize)]
26pub enum Stage {
27    Blinds,
28    Hole,
29    PreFlop,
30    Flop,
31    Turn,
32    River,
33    ShowDown,
34}
35/// Implementarion of Display trait for Stage.
36impl Display for Stage {
37    fn fmt(&self, f: &mut fmt::Formatter) -> fmt::Result {
38        match self {
39            Stage::Blinds => write!(f, "Blinds"),
40            Stage::Hole => write!(f, "Hole"),
41            Stage::PreFlop => write!(f, "Pre-Flop"),
42            Stage::Flop => write!(f, "Flop"),
43            Stage::Turn => write!(f, "Turn"),
44            Stage::River => write!(f, "River"),
45            Stage::ShowDown => write!(f, "Showdown"),
46        }
47    }
48}

102    pub fn build(big_blind: usize, max_players: u8) -> Self {
103        let mut game = Game {
104            players: HashMap::new(),
105            players_order: Vec::new(),
106            dealer: None,
107            buy_in: 100 * big_blind,
108            small_blind: big_blind / 2,
109            big_blind,
110            pot: 0,
111            side_pots: Vec::new(),
112            deck: Vec::new(),
113            community_cards: Vec::new(),
114            max_players,
115            winner: None,
116            stage: Stage::Blinds,
117            num_rounds: 0,
118            uuid: Uuid::new_v4(),
119        };
120        if max_players > MAX_PLAYERS {
121            panic!("The maximum number of players is {}", MAX_PLAYERS);
122        }
123        if max_players < MIN_PLAYERS {
124            panic!("The minimum number of players is {}", MIN_PLAYERS);
125        }
126        let mut deck = card::new_deck();
127        let mut rng = rng();
128        deck.shuffle(&mut rng);
129        game.deck = deck;
130
131        game
132    }

134    /// Predicate function for the game having the full amount of players.
135    fn full(&self) -> bool {
136        let num_players = self
137            .max_players
138            .to_usize()
139            .expect("Could not convert num_players to usize");
140        self.players.len() == num_players
141    }

143    /// Allows a player to join the game. The new player's bank roll will be equal to the buy in amount.
144    /// The player's name may be changed to make it unique among existing players. The player
145    /// instance is notified of the name and bank roll via Player::set_name_and_bank_roll.
146    pub fn join(&mut self, mut player: Player) -> Result<(), &'static str> {
147        if self.full() {
148            return Err("Cannot add more players.");
149        }
150        let names: Vec<String> = self.players.values().map(|p| p.name.clone()).collect();
151        let name = names::uniquify_name(&player.name, &names);
152        player.set_name_and_bank_roll(&name, self.buy_in);
153        self.players.insert(name.clone(), Box::new(player));
154        self.players_order.push(name);
155        Ok(())
156    }

158    /// Play a game.
159    pub fn play(&mut self) -> Winner {
160        while self.players.len() > 1 {
161            self.play_round();
162            self.reset_after_round();
163            self.num_rounds += 1;
164        }
165        let w = self.get_winner();
166        let msg = Msg::GameWinner(w.clone());
167        self.update_players(&msg);
168        w
169    }

171    /// Determine the winner at the end of the game. Assumption is that there's only one
172    /// player left.
173    fn get_winner(&self) -> Winner {
174        let winner_opt = self.players_order.first();
175        if let Some(name) = winner_opt {
176            let winner = self.players.get(name).unwrap();
177            let mut cards = self.community_cards.clone();
178            let (h1, h2) = winner.hole.unwrap();
179            cards.push(h1);
180            cards.push(h2);
181            let name = winner.name.clone();
182            let hand = compare::best_hand(&cards);
183            Winner::SoleWinner(PlayerHand { name, hand, cards })
184        } else {
185            panic!("Announcing winner but they have been removed...")
186        }
187    }

208    /// Set the name of the dealer and reorder the players_order list
209    /// so that the player to the left of the dealer is at the front.
210    fn order_players(&mut self) {
211        if self.stage == Stage::Blinds {
212            let players_order: Vec<String> = self.players_order.clone();
213            self.dealer = Some(players_order.first().unwrap().clone());
214        }
215        let dealer = self.dealer.as_ref();
216        if self.players.contains_key(dealer.unwrap()) {
217            let pos = self
218                .players_order
219                .iter()
220                .position(|n| n == dealer.unwrap())
221                .unwrap();
222            let current_i = pos + 1 % self.players_order.len();
223            self.players_order = rotate_vector(&self.players_order, current_i);
224        } else {
225            self.dealer = Some(self.players_order.first().unwrap().clone());
226        }
227    }

229    /// Play a round.
230    fn play_round(&mut self) {
231        self.order_players();
232        self.ante_up();
233        self.announce_players();
234        self.stage = Stage::Hole;
235        self.deal_hole_cards();
236        self.stage = Stage::PreFlop;
237        self.place_bets();
238        self.stage = Stage::Flop;
239        self.deal_flop();
240        self.place_bets();
241        self.stage = Stage::Turn;
242        self.deal_turn();
243        self.place_bets();
244        self.stage = Stage::River;
245        self.deal_river();
246        self.place_bets();
247        self.stage = Stage::ShowDown;
248        self.showdown();
249        self.distribute_pots();
250        // announce the winner.
251        self.announce_winner_round();
252    }

254    /// Each player pays the small or big blind at the beginning of each round,
255    /// or chooses not to if they don't want to take part in this round.
256    /// The player to the left of the dealer pays the small blind, everyone else
257    /// pays the big blind.
258    fn ante_up(&mut self) {
259        if self.players_order.is_empty() {
260            return;
261        }
262
263        let players_order = self.players_order.clone();
264
265        // Handle the first player (Small Blind) in a restricted scope.
266        {
267            let left_of_dealer: &String = players_order.first().unwrap();
268
269            // Mutably borrow the player and execute the action
270            if let Some(first_p) = self.players.get_mut(left_of_dealer)
271                && let Some(blind) = first_p.ante_up(self.small_blind)
272            {
273                self.pot += blind;
274            }
275            // NB: player marks themself as folded if they responded negatively
276            // or as all in if their bank roll was less than the blind.
277        }
278
279        // Handle the remaining players.
280        players_order[1..].iter().for_each(|name| {
281            if let Some(p) = self.players.get_mut(name)
282                && let Some(blind) = p.ante_up(self.big_blind)
283            {
284                self.pot += blind
285            }
286        });
287    }

289    /// Each player receives two hole cards after ante up.
290    fn deal_hole_cards(&mut self) {
291        let mut deck = self.deck.clone();
292        self.players.iter_mut().for_each(|(_, p)| {
293            let c1 = deck.pop().unwrap();
294            let c2 = deck.pop().unwrap();
295            p.hole_cards((c1, c2));
296        });
297        self.deck = deck;
298    }

300    /// Burn one card and deal the first three three community cards.
301    fn deal_flop(&mut self) {
302        let mut deck = self.deck.clone();
303        let _burn = deck.pop().unwrap();
304        let c1 = deck.pop().unwrap();
305
306        let c2 = deck.pop().unwrap();
307        let c3 = deck.pop().unwrap();
308        self.community_cards.push(c1);
309        self.community_cards.push(c2);
310        self.community_cards.push(c3);
311        self.deck = deck;
312    }

332    /// Players are given the opportunity to bet. If a player raises the bet, every
333    /// other player must call or raise again.
334    fn place_bets(&mut self) {
335        // names of players who have not folded
336        let not_folded: Vec<(String, bool)> = self
337            .players
338            .values()
339            .filter(|p| !p.folded)
340            .map(|p| (p.name.clone(), p.all_in))
341            .collect();
342        // names of players who have not folded and are not all in. These are the players who need to make a bet/call/fold.
343        let mut not_all_in: Vec<String> = not_folded
344            .iter()
345            .filter(|(_name, all_in)| !all_in)
346            .map(|(name, _all_in)| name.clone())
347            .collect();
348        // The players who will be betting, in the right order
349        let mut players: Vec<String> = Vec::new();
350        for name in self.players_order.clone() {
351            if not_folded.iter().any(|(n, _b)| n == &name) {
352                players.push(name);
353            }
354        }
355        if players.is_empty() {
356            return;
357        }
358
359        let update = Msg::StageDeclare(self.stage, self.community_cards.clone());
360        self.update_players(&update);
361
362        let mut current_index: usize = 0;
363        // target is the player at which the round of betting should stop.
364        // This is altered when a player raises the bet.
365        let mut target: String = players.first().unwrap().clone();
366        let mut done: bool = false; // flag to stop the loop
367        let mut target_placed_bet: bool = false; // flag to allow target to place first bet.
368        let mut call: usize = 0;
369        let min = self.big_blind;
370        let mut cycle: u8 = 0; // the number of times players have been given a chance to bet in this round.
371
372        // Ask each player to place a bet at least once. Note that the Player struct is responsible
373        // for managing its own state during betting, e.g. keeping the bank roll up to date
374        // and whether the player is folded or all in.
375        while !done && players.len() > 1 {
376            let current_name = &players[current_index % players.len()];
377            let p = self.players.get_mut(current_name).unwrap();
378            if p.name == target && target_placed_bet {
379                done = true;
380            } else {
381                if p.name == target {
382                    target_placed_bet = true;
383                }
384                if !p.all_in && !p.folded {
385                    let ccards = self.community_cards.clone();
386                    let args = BetArgs {
387                        call,
388                        min,
389                        stage: self.stage,
390                        cycle,
391                        community_cards: ccards,
392                    };
393                    let bet_opt = p.place_bet(args);
394
395                    let bet = bet_opt.unwrap();
396
397                    match bet {
398                        Bet::Fold => {
399                            players.remove(current_index);
400                            continue; // continue without incrementing current
401                        }
402                        Bet::Check => {
403                            if call > 0 {
404                                panic!(
405                                    "Misbehaving client checked when there was an outstanding bet."
406                                );
407                            }
408                        }
409                        Bet::Call => {
410                            self.pot += call;
411                        }
412                        Bet::Raise(raise) => {
413                            cycle += 1;
414                            if !self.side_pots.is_empty() {
415                                let side_pot = self.side_pots.get_mut(0).unwrap();
416                                side_pot.pot += raise;
417                            } else {
418                                self.pot += raise;
419                            }
420                            // raise is the new amount to match/beat
421                            call = raise;
422                            target = p.name.clone();
423                        }
424                        Bet::AllIn(bet) => {
425                            self.pot += bet;
426
427                            if let Some(index) =
428                                not_all_in.iter().position(|value| value == &p.name)
429                            {
430                                not_all_in.swap_remove(index);
431                            }
432
433                            let new_side_pot = SidePot {
434                                players: not_all_in.clone(),
435                                pot: 0,
436                            };
437                            self.side_pots.push(new_side_pot);
438                            // don't ask this player again in this round.
439                            players.remove(current_index);
440                            continue; // continue without incrementing current
441                        }
442                    }
443                    let update = Msg::Bet {
444                        player: p.name.clone(),
445                        bet,
446                        pot: self.pot,
447                    };
448                    self.update_players(&update);
449                }
450                current_index = (current_index + 1) % players.len();
451            }
452        }
453    }

462    /// Determines the winner(s) of the round.
463    fn showdown(&mut self) {
464        // Get the best hand for each non-folded player.
465        let mut hands: Vec<PlayerHand> =
466            self.names_to_hands(self.players.keys().cloned().collect());
467        // Handle cases where 0 or 1 players remain (the last player standing wins)
468        if hands.len() < 2 {
469            if let Some(PlayerHand {
470                name,
471                hand: best_hand,
472                cards,
473            }) = hands.pop()
474            {
475                let winner = Winner::SoleWinner(PlayerHand {
476                    name,
477                    hand: best_hand,
478                    cards,
479                });
480                self.winner = Some(winner);
481            } else {
482                panic!("No players remaining to determine winner.");
483            }
484            return;
485        }
486
487        let winner = Game::determine_winner(hands);
488        self.winner = Some(winner);
489    }

491    /// Takes a vector of player names and return a vector of (name, hand, cards),
492    /// where hand is their best hand and cards is their hole cards plus community cards.
493    /// Result contains only non-folded players.
494    fn names_to_hands(&self, names: Vec<String>) -> Vec<PlayerHand> {
495        // Calculate the best hand for each non-folded player.
496        let hands: Vec<PlayerHand> = self
497            .players
498            .iter() // Use iter() since we don't need to mutate Player state here
499            .filter_map(|(_, p)| {
500                // Only consider players who haven't folded
501                if p.folded || !names.contains(&p.name) {
502                    return None;
503                }
504                let (c1, c2) = p
505                    .hole
506                    .expect("Hole cards should be dealt before calling names_to_hands");
507                // Collect all 7 cards (2 hole + 5 community)
508                let mut all_cards = self.community_cards.clone();
509                all_cards.push(c1);
510                all_cards.push(c2);
511
512                let best_hand = compare::best_hand(&all_cards);
513                Some(PlayerHand {
514                    name: p.name.clone(),
515                    hand: best_hand,
516                    cards: all_cards,
517                })
518            })
519            .collect();
520        hands
521    }

523    /// Determine winner(s) from vector of (name, best_hand, cards) tuples.
524    fn determine_winner(mut hands: Vec<PlayerHand>) -> Winner {
525        // Handle cases where 0 or 1 players remain (the last player standing wins)
526        if hands.len() < 2 {
527            if let Some(PlayerHand {
528                name,
529                hand: best_hand,
530                cards,
531            }) = hands.pop()
532            {
533                return Winner::SoleWinner(PlayerHand {
534                    name,
535                    hand: best_hand,
536                    cards,
537                });
538            } else {
539                panic!("No players remaining to determine winner.");
540            }
541        }
542
543        // Compare the hands.
544
545        // Initialize winner with the first player's hand.
546        let mut winner: Winner = {
547            let PlayerHand {
548                name,
549                hand: best_hand,
550                cards,
551            } = hands.remove(0);
552            Winner::SoleWinner(PlayerHand {
553                name,
554                hand: best_hand,
555                cards,
556            })
557        };
558
559        // Compare current winner against all remaining hands.
560        for PlayerHand {
561            name: challenger_name,
562            hand: challenger_hand,
563            cards: challenger_cards,
564        } in hands
565        {
566            winner = match winner {
567                Winner::SoleWinner(PlayerHand {
568                    name: w_name,
569                    hand: w_hand,
570                    cards: w_cards,
571                }) => {
572                    // Compare the current winner (w_...) against the challenger (challenger_...)
573                    compare::compare_hands(
574                        PlayerHand {
575                            name: challenger_name,
576                            hand: challenger_hand,
577                            cards: challenger_cards,
578                        },
579                        PlayerHand {
580                            name: w_name,
581                            hand: w_hand,
582                            cards: w_cards,
583                        },
584                    )
585                }
586                Winner::Draw(mut draw_winners) => {
587                    // It's a draw, compare the challenger against the best hand in the draw group.
588                    // Assume the first element of a Draw is the benchmark.
589                    let PlayerHand {
590                        name: w_name_benchmark,
591                        hand: w_hand_benchmark,
592                        cards: w_cards_benchmark,
593                    } = draw_winners.pop().unwrap();
594
595                    let comparison_result = compare::compare_hands(
596                        PlayerHand {
597                            name: challenger_name.clone(),
598                            hand: challenger_hand,
599                            cards: challenger_cards.clone(),
600                        },
601                        PlayerHand {
602                            name: w_name_benchmark.clone(),
603                            hand: w_hand_benchmark,
604                            cards: w_cards_benchmark.clone(),
605                        },
606                    );
607
608                    // Put the benchmark hand back for future comparisons or draw outcome
609                    draw_winners.push(PlayerHand {
610                        name: w_name_benchmark,
611                        hand: w_hand_benchmark,
612                        cards: w_cards_benchmark,
613                    });
614
615                    match comparison_result {
616                        Winner::SoleWinner(PlayerHand {
617                            name: n,
618                            hand: h,
619                            cards: c,
620                        }) => {
621                            if n == challenger_name {
622                                // Challenger is better than the benchmark (and thus all previous winners)
623                                Winner::SoleWinner(PlayerHand {
624                                    name: n,
625                                    hand: h,
626                                    cards: c,
627                                })
628                            } else {
629                                // Challenger is worse than the benchmark, keep the existing draw group
630                                Winner::Draw(draw_winners)
631                            }
632                        }
633                        Winner::Draw(_) => {
634                            // Challenger ties with the benchmark, add challenger to the draw group.
635                            // The original (un-cloned) challenger values are now moved here.
636                            draw_winners.push(PlayerHand {
637                                name: challenger_name,
638                                hand: challenger_hand,
639                                cards: challenger_cards,
640                            });
641                            Winner::Draw(draw_winners)
642                        }
643                    }
644                }
645            };
646        }
647        winner
648    }

650    /// Distributes the pot and side pot to the winner(s).
651    ///
652    /// TODO
653    /// + refactor this into several smaller methods,
654    /// + keep track of chips being lost due to truncating division.
655    fn distribute_pots(&mut self) {
656        let winner = self.winner.clone();
657        let main_pot = self.pot;
658        let side_pots = self.side_pots.clone();
659        let ccards = self.community_cards.clone();
660        // details of not folded players: names, bests hands, full sets of cards and whether they are all in
661        let not_folded: Vec<(PlayerHand, bool)> = self
662            .players
663            .values()
664            .filter(|p| !p.folded)
665            .map(|p| {
666                let (c1, c2) = p.hole.unwrap();
667                let mut cards = ccards.clone();
668                cards.extend(vec![c1, c2]);
669                (
670                    PlayerHand {
671                        name: p.name.clone(),
672                        hand: compare::best_hand(&cards),
673                        cards,
674                    },
675                    p.all_in,
676                )
677            })
678            .collect();
679        // not folded and not all in
680        let not_all_in: Vec<PlayerHand> = not_folded
681            .iter()
682            .filter(|(_ph, all_in)| !all_in)
683            .map(
684                |(
685                    PlayerHand {
686                        name,
687                        hand: best_hand,
688                        cards,
689                    },
690                    _all_in,
691                )| PlayerHand {
692                    name: name.clone(),
693                    hand: *best_hand,
694                    cards: cards.clone(),
695                },
696            )
697            .collect();
698        let not_folded_clone = not_folded.clone();
699        // store winnings during distribution algorithm, allocate at end
700        let mut winnings: HashMap<String, usize> = HashMap::new();
701        for (ph, _b) in not_folded {
702            winnings.insert(ph.name, 0);
703        }
704        if let Some(w) = winner {
705            match w {
706                Winner::SoleWinner(PlayerHand { name, .. }) => {
707                    let winner_name = name.clone();
708                    if not_folded_clone.iter().any(|(ph, _all_in)| ph.name == name) {
709                        // winner is not folded
710                        // distribute the main pot
711                        match winnings.get_mut(&winner_name) {
712                            Some(value) => *value += main_pot, // If the key exists, update the value
713                            None => {
714                                panic!("Key '{}' does not exist in the HashMap.", winner_name)
715                            }
716                        }
717                        if not_all_in.iter().any(|ph| ph.name == winner_name) {
718                            // winner is not all in, they win the side pots too
719                            let side_pots: usize = self.side_pots.iter().map(|sp| sp.pot).sum();
720                            match winnings.get_mut(&winner_name) {
721                                Some(value) => *value += side_pots, // If the key exists, update the value
722                                None => {
723                                    panic!("Key '{}' does not exist in the HashMap.", winner_name)
724                                }
725                            }
726                        } else {
727                            // winner is all in, they only win side pots they contributed to
728                            // distribute side pots
729                            for sp in side_pots {
730                                // possible winners
731                                let candidates: Vec<PlayerHand> = not_folded_clone
732                                    .iter()
733                                    .filter(|(ph, _all_in)| sp.players.contains(&ph.name))
734                                    .map(|(ph, _all_in)| PlayerHand {
735                                        name: ph.name.clone(),
736                                        hand: ph.hand,
737                                        cards: ph.cards.clone(),
738                                    })
739                                    .collect();
740                                if candidates.is_empty() {
741                                    // everyone in this side pot has folded so the winnings go to the winner of the main pot
742                                    match winnings.get_mut(&winner_name) {
743                                        Some(value) => *value += sp.pot, // If the key exists, update the value
744                                        None => {
745                                            panic!("Key '{}' does not exist in the HashMap.", name)
746                                        }
747                                    }
748                                } else {
749                                    // players who participated in this side pot are still in the round
750                                    let w = Game::determine_winner(candidates);
751                                    match w {
752                                        // single winner for this side pot
753                                        Winner::SoleWinner(PlayerHand { name, .. }) => {
754                                            match winnings.get_mut(&name) {
755                                                Some(value) => *value += sp.pot, // If the key exists, update the value
756                                                None => panic!(
757                                                    "Key '{}' does not exist in the HashMap.",
758                                                    name
759                                                ),
760                                            }
761                                        }
762                                        // multiple winners for this side pot
763                                        Winner::Draw(winners) => {
764                                            let pot_share = sp.pot / winners.len();
765                                            for PlayerHand {
766                                                name,
767                                                hand: _,
768                                                cards: _,
769                                            } in winners
770                                            {
771                                                match winnings.get_mut(&name) {
772                                                    Some(value) => *value += pot_share, // If the key exists, update the value
773                                                    None => panic!(
774                                                        "Key '{}' does not exist in the HashMap.",
775                                                        name
776                                                    ),
777                                                }
778                                            }
779                                        }
780                                    }
781                                }
782                            }
783                        }
784                    } else {
785                        panic!("Winner not in not_folded.");
786                    }
787                }
788                Winner::Draw(winners) => {
789                    // distribute main pot
790                    let main_pot_share = main_pot / winners.len();
791                    for PlayerHand {
792                        name,
793                        hand: _,
794                        cards: _,
795                    } in winners.clone()
796                    {
797                        match winnings.get_mut(&name) {
798                            Some(value) => *value += main_pot_share,
799                            None => panic!("Key '{}' does not exist in the HashMap.", name),
800                        }
801                    }
802                    //distribute side pots
803                    for sp in side_pots {
804                        // possible winners
805                        let candidates: Vec<PlayerHand> = not_folded_clone
806                            .iter()
807                            .filter(|(ph, _all_in)| sp.players.contains(&ph.name))
808                            .map(|(ph, _all_in)| PlayerHand {
809                                name: ph.name.clone(),
810                                hand: ph.hand,
811                                cards: ph.cards.clone(),
812                            })
813                            .collect();
814                        if candidates.is_empty() {
815                            // everyone who contributed to this side pot has folded, the winners share the pot
816                            for PlayerHand {
817                                name,
818                                hand: _,
819                                cards: _,
820                            } in winners.clone()
821                            {
822                                match winnings.get_mut(&name) {
823                                    Some(value) => *value += sp.pot,
824                                    None => {
825                                        panic!("Key '{}' does not exist in the HashMap.", name)
826                                    }
827                                }
828                            }
829                        } else {
830                            // there are unfolded players who contributed to this side pot
831                            let w = Game::determine_winner(candidates);
832                            match w {
833                                // single winner for this side pot
834                                Winner::SoleWinner(PlayerHand { name, .. }) => {
835                                    match winnings.get_mut(&name) {
836                                        Some(value) => *value += sp.pot,
837                                        None => {
838                                            panic!("Key '{}' does not exist in the HashMap.", name)
839                                        }
840                                    }
841                                }
842                                // multiple winners for this side pot
843                                Winner::Draw(winners) => {
844                                    let pot_share = sp.pot / winners.len();
845                                    for PlayerHand { name, .. } in winners {
846                                        match winnings.get_mut(&name) {
847                                            Some(value) => *value += pot_share,
848                                            None => panic!(
849                                                "Key '{}' does not exist in the HashMap.",
850                                                name
851                                            ),
852                                        }
853                                    }
854                                }
855                            }
856                        }
857                    }
858                }
859            }
860            // distribute winnings
861            for (name, pot_share) in winnings.clone() {
862                if pot_share > 0 {
863                    match self.players.get_mut(&name) {
864                        Some(value) => value.bank_roll += pot_share,
865                        None => panic!("Key '{}' does not exist in the HashMap.", name),
866                    }
867                }
868            }
869            self.pot = 0;
870            self.side_pots = Vec::new();
871        } else {
872            dbg!("Distribute pots called with no winner set.");
873        }
874    }

876    /// Reset the Game and Players after a round.
877    fn reset_after_round(&mut self) {
878        self.pot = 0;
879        self.side_pots = Vec::new();
880        self.community_cards = Vec::new();
881        self.deck = card::new_deck();
882        let dealer_name_ref: Option<&String> = self.dealer.as_ref();
883
884        let mut removed_names: Vec<String> = Vec::new();
885
886        // Loop through the players resetting all_in and folded, and collecting
887        // the list of ones that need to be removed for lack of money.
888        for name in self.players_order.iter() {
889            let p = self
890                .players
891                .get_mut(name)
892                .expect("Player in order list not found in map.");
893            p.all_in = false;
894            p.folded = false;
895            p.hole = None;
896            let is_dealer = dealer_name_ref.unwrap() == name;
897            // if player doesn't have enough chips to continue, mark for removal
898            if (is_dealer && p.bank_roll < self.small_blind) || (p.bank_roll < self.big_blind) {
899                removed_names.push(name.clone());
900            }
901        }
902
903        // remove player names from self.players_order and Player structs from self.player
904        for name in removed_names.iter() {
905            self.players.remove(name.as_str());
906            if let Some(index) = self.players_order.iter().position(|value| value == name) {
907                self.players_order.swap_remove(index);
908            }
909            self.players.remove(name);
910        }
911        // Add players to bring up the numbers?
912    }

1036        if bank_roll == 0 {
1037            Bet::Fold
1038        } else if bank_roll <= args.call {
1039            Bet::AllIn(bank_roll)

1041            Bet::Raise(args.min)
1042        } else {
1043            Bet::Call
1044        }
