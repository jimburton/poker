1/// Functions for generating and uniquifying names.
2use rand::Rng;
3use std::collections::HashSet;
4
5// Common names from around the world.
6const NAMES: [&str; 12] = [
7    "Bob", "Alice", "Cali", "Arjun", "Bianca", "Kalyna", "Chen", "Zhu", "Cielo", "Eva", "Franco",
8    "Lopa",
9];
10
11/// Return n distinct names, where n is up to the length of NAMES.
12pub fn get_names(n: usize) -> Result<Vec<String>, &'static str> {
13    if n > NAMES.len() {
14        return Err("Request a smaller number of names");
15    }
16
17    let mut rng = rand::rng();
18    // create n random indices.
19    let mut indices: Vec<u8> = Vec::new();
20    while indices.len() < n {
21        let i = rng.random_range(0..n) as u8;
22        if !indices.contains(&i) {
23            indices.push(i);
24        }
25    }
26    Ok(indices
27        .iter()
28        .map(|i| NAMES[*i as usize].to_string())
29        .collect())
30}
31
32/// Modify the incoming list to make them distinct.
33#[allow(clippy::ptr_arg)]
34pub fn uniquify(names: &Vec<String>) -> Vec<String> {
35    let mut names = names.clone();
36    let mut names_set: HashSet<String> = HashSet::from_iter(names.iter().cloned());
37    if names.len() == names_set.len() {
38        return names;
39    }
40    // Add random digits to the end of names until the list contains only distinct values.
41    let mut rng = rand::rng();
42    for i in 0..names.len() - 1 {
43        if names[(i + 1)..].contains(&names[i]) {
44            let d = rng.random_range(0..10).to_string();
45            names[i] = names[i].clone() + &d;
46        }
47    }
48    names_set = HashSet::from_iter(names.iter().cloned());
49    if names.len() == names_set.len() {
50        names
51    } else {
52        uniquify(&names)
53    }
54}
55
56/// Modify name to make it distinct with respect to names.
57#[allow(clippy::ptr_arg)]
58pub fn uniquify_name(name: &String, names: &Vec<String>) -> String {
59    if !names.contains(name) {
60        return name.to_owned();
61    }
62    // Add a random digit to the end of name.
63    let mut rng = rand::rng();
64    let d = rng.random_range(0..10).to_string();
65    let name_plus = name.clone() + &d;
66    if !names.contains(&name_plus) {
67        name_plus
68    } else {
69        uniquify_name(&name_plus, names)
70    }
71}

77    #[test]
78    fn test_get_names() {
79        let six_names_result = get_names(6);
80        assert!(
81            six_names_result.is_ok(),
82            "Should be able to retrieve six names."
83        );
84        let six_names = six_names_result.unwrap();
85        assert!(
86            six_names.len() == 6,
87            "Expected six_names.len() == 6, was {}",
88            six_names.len()
89        );
90        let max_names_result = get_names(NAMES.len());
91        assert!(
92            max_names_result.is_ok(),
93            "Should be able to retrieve the max number of names ({}).",
94            NAMES.len()
95        );
96        let max_names = max_names_result.unwrap();
97        assert!(
98            max_names.len() == NAMES.len(),
99            "Expected max_names.len() == {}, was {}",
100            NAMES.len(),
101            max_names.len()
102        );
103        max_names.iter().for_each(|name| {
104            assert!(
105                NAMES.contains(&&name[..]),
106                "Expected NAMES to contain {}",
107                name
108            )
109        });
110        let too_many_names_result = get_names(NAMES.len() + 1);
111        assert!(
112            too_many_names_result.is_err(),
113            "Should not be able to retrieve more than max number of names ({}).",
114            NAMES.len()
115        );
116        let no_names_result = get_names(0);
117        assert!(
118            no_names_result.is_ok(),
119            "Should be able to retrieve zero names."
120        );
121        let no_names = no_names_result.unwrap();
122        assert!(
123            no_names.is_empty(),
124            "Expected no_names to be empty, length was {}",

129    #[test]
130    fn test_uniquify_names() {
131        let dups = vec!["a".to_string(), "a".to_string()];
132        let result = uniquify(&dups);
133        assert!(
134            dups.len() == result.len(),
135            "Expected result to have same length ({}), was {}",
136            dups.len(),
137            result.len()
138        );
139        let uniqs: HashSet<String> = HashSet::from_iter(result.iter().cloned());
140        assert!(
141            dups.len() == uniqs.len(),
142            "Expected unique names to have same length ({}) as result, was {}",
143            result.len(),
144            uniqs.len()
145        );
146        let uniqs = vec!["a".to_string(), "b".to_string()];
147        let result = uniquify(&uniqs);
148        (0..2).for_each(|i| {
149            assert!(
150                uniqs[i] == result[i],
151                "Expected function to be stable, found {} in place of {}",
152                result[i],
153                uniqs[i]
154            )
155        });
156        let many_dups = vec![
157            "Bob".to_string(),
158            "Bob".to_string(),
159            "Bob".to_string(),

161            "Bob".to_string(),
162            "Bob".to_string(),
163        ];
164        let result = uniquify(&many_dups);
165        assert!(
166            many_dups.len() == result.len(),

168            many_dups.len(),
169            result.len()

171        let uniqs: HashSet<String> = HashSet::from_iter(result.iter().cloned());
172        assert!(
