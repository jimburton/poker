20    /// Place a bet.
21    fn place_bet(
22        &mut self,
23        args: BetArgs,
24        hole_cards: (Card, Card),
25        bank_roll: usize,
26    ) -> Option<Bet> {
27        let mut cards = args.community_cards.clone();
28        let (h1, h2) = (hole_cards.0, hole_cards.1);
29        cards.push(h1);
30        cards.push(h2);
31        let bh = best_hand(&cards);
32
33        println!("It's your turn to place a bet in the {}.", args.stage);
34        println!("Hole cards: {}, {}", h1, h2);
35        if !args.community_cards.is_empty() {
36            println!("Community cards:",);
37            args.community_cards.iter().for_each(|c| println!("{}", c));
38        }
39        println!(
40            "The bet stands at {} (minimum amount to bet {})",
41            args.call, args.min
42        );
43        println!("Bank roll: {}. Best hand: {}", bank_roll, bh);
44        println!("Enter R(aise) <amount>, C(all), Ch(eck), A(ll in), F(old)");
45        let mut input = String::new(); // A mutable String to hold the user input
46        std::io::stdin()
47            .read_line(&mut input) // Read input into the `input` variable
48            .expect("Failed to read line");
49
50        if let Some(bet) = parse_bet_string(input, bank_roll) {
51            match bet {
52                Bet::Fold => Some(Bet::Fold),
53                Bet::Check => Some(Bet::Check),
54                Bet::Call => Some(Bet::Call),
55                Bet::Raise(n) => Some(Bet::Raise(n)),
56                Bet::AllIn(n) => Some(Bet::AllIn(n)),
57            }
58        } else {
59            None
60        }
61    }
62
63    fn update(&self, msg: &Msg) {
64        match msg {
65            Msg::Player { name, bank_roll } => {
66                println!("Playing as {} ({})", name, bank_roll);
67            }
68            Msg::HoleCards { cards } => {
69                println!("Received hole cards: {}, {}", cards.0, cards.1);
70            }
71            Msg::Bet { player, bet, pot } => {
72                println!("Player {} made bet: {} (pot is now {})", player, bet, pot);
73            }
74            Msg::PlayersInfo { players, dealer } => {
75                println!(
76                    "Playing this round with: {}",
77                    players
78                        .iter()
79                        .map(|(player_name, bank_roll)| {
80                            let mut name = player_name.clone();
81                            if name == *dealer {
82                                name += " [Dealer]";
83                            }
84                            name + " (" + &bank_roll.to_string() + ")"
85                        })
86                        .collect::<Vec<String>>()
87                        .join(", "),
88                );
89            }
90            Msg::GameWinner(w) => {
91                println!("##############\n## {}.\n##############", w,);
92            }
93            Msg::RoundWinner(w) => {
94                println!("##############\n## {}.\n##############", w,);
95            }
96            Msg::StageDeclare(stage, _community_cards) => {
97                println!(
98                    "##############\n## The {} stage is beginning.\n##############",
99                    stage,
100                );
101            }
102        }
103    }
104}

106fn parse_bet_string(input: String, all_in_amount: usize) -> Option<Bet> {
107    let parts: Vec<&str> = input.trim().split(" ").collect();
108    if parts.len() == 2 {
109        let amount: usize = parts[1]
110            .trim() // Remove whitespace
111            .parse() // Convert to i32
112            .expect("Please enter a valid number");
113        Some(Bet::Raise(amount))
114    } else {
115        match parts[0] {
116            "C" => Some(Bet::Call),
117            "Ch" => Some(Bet::Check),
118            "F" => Some(Bet::Fold),
119            "A" => Some(Bet::AllIn(all_in_amount)),
120            _ => None,
121        }
122    }
123}
