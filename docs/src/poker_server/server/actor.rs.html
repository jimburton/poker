20const CHANNEL_CAPACITY: usize = 32;

22type BetRequest = (BetArgs, (Card, Card), usize, oneshot::Sender<Option<Bet>>);

65/// The thread-safe, cloneable structure used by the synchronous facade
66/// to push messages into the asynchronous WebSocket task.
67#[derive(Debug, Clone)]
68struct RemoteActorHandle {
69    // Channel sender for sending updates (Msg) to the WebSocket loop
70    update_tx: mpsc::Sender<PokerMessage>,
71    // Channel sender for requesting a bet (place_bet) and receiving the result
72    // The player thread sends a request to the loop, and the loop replies here.
73    bet_tx: mpsc::Sender<BetRequest>,
74}
75/// The actual asynchronous loop that manages the single, non-cloneable WebSocket.
76async fn start_socket_loop(
77    mut socket: WebSocket,
78    mut update_rx: mpsc::Receiver<PokerMessage>,
79    mut bet_rx: mpsc::Receiver<BetRequest>,
80) {
81    // Run the loop indefinitely until the socket closes or an error occurs
82    loop {
83        tokio::select! {
84            // 1. Handle incoming messages from the synchronous game loop (Updates and Bet Requests)
85            // This is how the game engine tells the player to do something.
86            Some((args, hole_cards, bank_roll, bet_responder)) = bet_rx.recv() => {
87                // When the game engine calls place_bet, it sends a oneshot channel here.
88
89                // Construct the bet request message
90                let mut cards = args.community_cards.clone();
91                let (h1, h2) = (hole_cards.0, hole_cards.1);
92                cards.push(h1);
93                cards.push(h2);
94                let bh = best_hand(&cards);
95                let bet_msg = PokerMessage::PlaceBet {
96                    args,
97                    hole_cards,
98                    bank_roll,
99                    best_hand: bh,
100                };
101
102                // Send the request to the client
103                let send_res = socket
104                    .send(Message::Text(Utf8Bytes::from(
105                        serde_json::to_string(&bet_msg).unwrap(),
106                    )))
107                    .await;
108
109                if send_res.is_err() {
110                    let _ = bet_responder.send(None);
111                    return; // Exit loop on send error
112                }
113
114                // Wait for the client's response (this is the actual blocking network IO)
115                if let Some(msg) = socket.recv().await {
116                    match msg {
117                        Ok(Message::Text(utf8_bytes)) => {
118                println!("Received bytes: {:?}", utf8_bytes);
119                let bet = safe_deserialise::<PokerMessage>(&utf8_bytes);
120                println!("Deserialised as: {:?}", bet);
121                            // Extract the bet action
122                            let final_bet = match bet {
123                                Some(PokerMessage::PlayerBet(b)) => Some(b),
124                                _ => {
125                                    error!("Expected PlayerBet, got something else: {:?}", bet);
126                                    None
127                                }
128                            };
129                            let _ = bet_responder.send(final_bet);
130                        }
131                        _ => {
132                            eprintln!("Received non-text message or error during recv.");
133                            let _ = bet_responder.send(None);
134                        }
135                    }
136                } else {
137                    // Recv returned None, meaning socket closed.
138                    let _ = bet_responder.send(None);
139                    return;
140                }
141            }
142
143            // 2. Handle incoming updates from the synchronous game loop
144            Some(msg) = update_rx.recv() => {
145                // General updates are sent to the client (fire-and-forget)
146                let send_res = socket
147                    .send(Message::Text(Utf8Bytes::from(
148                        serde_json::to_string(&msg).unwrap(),
149                    )))
150                    .await;
151
152                if send_res.is_err() {
153                    return; // Exit loop on send error
154                }
155            }
156
157            // 3. Handle messages spontaneously sent from the client (e.g., chat)
158            Some(result) = socket.recv() => {
159                match result {
160                    Ok(Message::Text(_utf8_bytes)) => {
161                        // Handle unsolicited messages here (e.g., chat or keep-alive).
162                        // Do nothing, since we only care about bet responses during place_bet
163                    }
164                    Ok(Message::Close(_)) => {
165                        error!("WebSocket closed by client.");
166                        return;
167                    }
168                    Err(e) => {
169                        error!("WebSocket error: {}", e);
170                        return;
171                    }
172                    _ => {} // Ignore Binary, Ping, Pong, etc.
173                }
174            }
175
176            // Exit if all senders are dropped (no more messages to manage)
177            else => {
178                break;
179            }
180        }
181    }
182    // Clean up or log closure here
183}

185/// The Synchronous Facade struct that implements the Actor trait.
186#[derive(Debug)]
187pub struct RemoteActor {
188    runtime_handle: Handle,
189    handle: RemoteActorHandle,
190}
191impl RemoteActor {
192    /// Builds a new RemoteActor, starts the asynchronous WebSocket loop, and returns the facade.
193    pub fn build(socket: WebSocket, runtime_handle: Handle) -> RemoteActor {
194        // Create channels for communication between the facade and the async loop
195        let (update_tx, update_rx) = mpsc::channel(CHANNEL_CAPACITY);
196        let (bet_tx, bet_rx) = mpsc::channel(1); // Only need capacity 1 for blocking bets
197
198        // Start the continuous asynchronous task that owns the WebSocket
199        runtime_handle.spawn(start_socket_loop(socket, update_rx, bet_rx));
200
201        RemoteActor {
202            runtime_handle,
203            handle: RemoteActorHandle { update_tx, bet_tx },
204        }
205    }
206}
207// NOTE: The Actor trait methods must be defined here, NOT in the provided code snippet
208// I have created a simplified, conceptual impl for demonstration purposes.
209impl Actor for RemoteActor {
210    /// Accept the name and bank roll at the beginning of the game.
211    fn set_name_and_bank_roll(&self, name: &String, bank_roll: usize) {
212        let msg = Msg::Player {
213            name: name.clone(),
214            bank_roll,
215        };
216        self.update(&msg);
217    }
218
219    /// Accept the hole cards.
220    fn hole_cards(&self, hole_cards: (Card, Card)) {
221        let hole_card_msg = Msg::HoleCards { cards: hole_cards };
222        self.update(&hole_card_msg);
223    }
224    /// Place a bet (Synchronous, Blocking).
225    fn place_bet(
226        &mut self, // Changed to &self as state is shared via channels
227        args: BetArgs,
228        hole_cards: (Card, Card),
229        bank_roll: usize,
230    ) -> Option<Bet> {
231        println!("Sending bet request: {:?}", args.clone());
232        // Create a standard library blocking MPSC channel for the final result.
233        // This is safe for blocking the calling thread, even if it's a Tokio worker.
234        let (sync_tx, sync_rx) = std_mpsc::channel();
235
236        // Clone necessary parts to move into the blocking thread
237        let tx_clone = self.handle.bet_tx.clone();
238        let runtime_handle_clone = self.runtime_handle.clone();
239
240        // Spawn the entire request/response sequence onto a dedicated blocking thread.
241        runtime_handle_clone.spawn_blocking(move || {
242            // This runs on a separate thread pool where blocking is allowed.
243
244            // Tokio oneshot channel for receiving from the socket loop
245            let (result_tx, result_rx) = oneshot::channel();
246            let request: BetRequest = (args, hole_cards, bank_roll, result_tx);
247
248            // Send the request to the async loop (blocking_send is safe here)
249            if tx_clone.blocking_send(request).is_err() {
250                error!("Failed to send bet request to socket loop from blocking thread.");
251                let _ = sync_tx.send(None); // Send None back to the caller
252                return;
253            }
254
255            // Block the current dedicated thread until the async loop replies
256            let result = match result_rx.blocking_recv() {
257                Ok(bet) => bet,
258                Err(e) => {
259                    error!("Oneshot receiver error on blocking thread: {}", e);
260                    None
261                }
262            };
263
264            // d. Send the final result back to the original calling thread via std::mpsc
265            let _ = sync_tx.send(result);
266        });
267
268        // The thread calling `place_bet` blocks on the standard library receiver.
269        // This is the clean, safe way to block on a Tokio worker thread.
270        match sync_rx.recv() {
271            Ok(bet) => bet, // The Option<Bet> result
272            Err(e) => {
273                // The standard MPSC channel failed (e.g., sender dropped unexpectedly)
274                error!("Standard MPSC channel error waiting for result: {}", e);
275                None
276            }
277        }
278    }
279
280    /// Update (Synchronous, Non-Blocking).
281    fn update(&self, msg: &Msg) {
282        println!("Sending update: {}", msg);
283        // Convert the synchronous Msg into the asynchronous PokerMessage
284        let poker_msg = match msg {
285            Msg::Player { name, bank_roll } => PokerMessage::Player {
286                name: name.clone(),
287                bank_roll: *bank_roll,
288            },
289            Msg::HoleCards { cards } => PokerMessage::HoleCards { cards: *cards },
290            Msg::Bet { player, bet, pot } => PokerMessage::BetPlaced {
291                player: player.clone(),
292                bet: *bet,
293                pot: *pot,
294            },
295            Msg::PlayersInfo { players, dealer } => PokerMessage::PlayersInfo {
296                players: players.clone(),
297                dealer: dealer.clone(),
298            },
299            Msg::GameWinner(winner) => PokerMessage::GameWinner {
300                winner: winner.clone(),
301            },
302            Msg::RoundWinner(winner) => PokerMessage::RoundWinner {
303                winner: winner.clone(),
304            },
305            Msg::StageDeclare(stage, community_cards) => PokerMessage::StageDecl {
306                stage: *stage,
307                community_cards: community_cards.clone(),
308            },
309        };
310        let tx = self.handle.update_tx.clone();
311        self.runtime_handle.spawn(async move {
312            if let Err(e) = tx.send(poker_msg).await {
313                error!("Failed to send update message: {}", e);
314            }
315        });
316    }
